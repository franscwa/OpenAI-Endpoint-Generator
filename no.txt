import os

import openai
from flask import Flask, redirect, render_template, request, url_for

app = Flask(__name__)
openai.api_key = os.getenv("OPENAI_API_KEY")


@app.route("/", methods=("GET", "POST"))
def index():
    if request.method == "POST":
        #serverlang = request.form["serverlang"]
        #dbprovider = request.form["dbprovider"]
        endpoints = request.form["endpoints"]
        response = openai.Completion.create(
            model="text-davinci-002",
            prompt=generate_prompt(endpoints),
            temperature=0.6,
            max_tokens=1000,
        )
        return redirect(url_for("index", result=response.choices.text))

    result = request.args.get("result")
    return render_template("index.html", result=result)


def generate_prompt(endpoints):
    return """code a program using express with a postgres database with API endpoints
     for {}.

"""
